Private Function well_InitData(Q_m3Day, fw_perc, _
                      Optional Pcas_atma As Double = 10, _
                      Optional wellStr As String = WELL_DEFAULT, _
                      Optional PVTStr As String = PVT_DEFAULT, _
                      Optional ESPStr As String = ESP_DEFAULT, _
                      Optional HydrCorr As H_CORRELATION = 0, _
                      Optional Ksep_fr As Double = 0, _
                      Optional Kdegr_d As Double = 0) As CWell
    ' функция для шаблонного чтения данных по скважине в интерфейсных функциях
    '
    ' на входе данные по конструкции скважины, PVT, ЭЦН берутся из строк с закодированными параметрами
    '    следует использовать функции Encode для передачи параметров
    ' на выходе объект скважина с загруженными данными
    Dim well As New CWell
    Dim PVT As New CPVT
    Dim pmp As CESPpump
    Dim ESPsystem As New CESPsystemSimple
    
    Set PVT = PVT_Decode_string(PVTStr)
    PVT.Qliq_scm3day = Q_m3Day
    PVT.wc_fr = fw_perc / 100
    
    Set well = Well_Decode_string(wellStr)
    Set well.Fluid = PVT
    well.Pan_atma = Pcas_atma
    well.HFlowCorrelation = HydrCorr
     
    Set ESPsystem = ESP_Decode_string(ESPStr)
    Set well.ESP = ESPsystem
    well.ESP.ESPPressureDegradation = Kdegr_d
    If Ksep_fr > 0 And Ksep_fr <= 1 Then
        Call well.setKseptotal_d(valueManual, Ksep_fr, 0)
    Else
        Call well.setKseptotal_d(byCorrealation, Ksep_fr)
    End If
        
    Set well_InitData = well
End Function
Private Function Well_out_arr(well As CWell, Optional FirsrCol As Integer = 0, Optional type_out As Integer = 0)
    '14.04.2019
    ' type_out = 0 - стандартный вывод параметров,
    ' type_out = 1 - вывод параметров для анализа УРЭ,
    ' type_out = 2 - вывод только КПД УЭЦН, считается здесь же,
    ' все массивы заполняются параллельно
    Dim ar1, ar2
    Dim eff_ESP_total_d As Double
    
    With well
        ' подготовим данные для вывода
        ' данные выводятся в одну линию, чтобы, хотя бы теоретически, можно было вывести данные по скважине в таблице
        ' во второй строке выводятся подписи параметров, если необходимо
        ReDim ar1(20)
        ReDim ar2(20)
        ReDim ar3_power(19)
        ReDim ar4_names_power(19)
        eff_ESP_total_d = .ESP.ESPpump.PowerFluid_Wt / .ESP.Power_CS_teor_calc_W
        Dim i As Integer
        ' первый параметр настраиваемый
        i = 0
        ar1(0) = "":  ar2(0) = ""
        ar3_power(0) = "":  ar4_names_power(0) = ""
        ' блок параметров по давлениям
        i = i + 1
        ar1(i) = .Pline_atma:  ar2(i) = "Pline_atma"
        ar3_power(i) = .ESP.Power_shaft_W:  ar4_names_power(i) = "Power_shaft_W"
        i = i + 1
        ar1(i) = .Pbuf_atma:  ar2(i) = "Pbuf_atma"
        ar3_power(i) = .ESP.Power_motor_W:  ar4_names_power(i) = "Power_motor_W"
        i = i + 1
        ar1(i) = .Pdis_atma:  ar2(i) = "Pdis_atma"
        ar3_power(i) = .ESP.cable_dPower_W:  ar4_names_power(i) = "cable_dPower_W"
        i = i + 1
        ar1(i) = .Pint_atma:  ar2(i) = "Pint_atma"
        ar3_power(i) = .ESP.cable_Power_W: ar4_names_power(i) = "cable_Power_W"
        i = i + 1
        ar1(i) = .Pwf_atma:  ar2(i) = "Pwf_atma"
        ar3_power(i) = .ESP.dPower_transform_W: ar4_names_power(i) = "dPower_transform_W"
        i = i + 1
        ar1(i) = .Pan_atma:   ar2(i) = "Pan_atma"
        ar3_power(i) = .ESP.dPower_CS_W: ar4_names_power(i) = "dPower_CS_W"
        ' динамический уровень
        i = i + 1
        ar1(i) = .Hdyn_m:   ar2(i) = "Hdyn_m"
        ar3_power(i) = .ESP.Power_CS_teor_calc_W: ar4_names_power(i) = "Power_CS_teor_calc_W"
        ' электрические параметры
        i = i + 1
        ar1(i) = .ESP.Power_CS_calc_W:  ar2(i) = ".Power_W"
        ar3_power(i) = .ESP.Power_CS_calc_W: ar4_names_power(i) = "Power_CS_calc_W"
        i = i + 1
        ar1(i) = .ESP.freq_Hz:  ar2(i) = "freq_Hz"
        ar3_power(i) = .ESP.PowNom_corr_W: ar4_names_power(i) = "PowNom_corr_W"
        i = i + 1
        ar1(i) = .ESP.I_A:  ar2(i) = "I_A"
        ar3_power(i) = .ESP.Load_fr: ar4_names_power(i) = "Load_fr"
        i = i + 1
        ar1(i) = .ESP.Load_fr:  ar2(i) = "Load_fr"
        ar3_power(i) = .ESP.motor_eff_fr: ar4_names_power(i) = "motor_eff_fr"
        i = i + 1
        ar1(i) = .ESP.ESPpump.EffiencyESP_d:  ar2(i) = "EffESP_d"
        ar3_power(i) = .ESP.CS_eff_fr: ar4_names_power(i) = "CS_eff_fr"
        ' параметры сепарации
        i = i + 1
        ar1(i) = .ESP.KsepTotal_fr:  ar2(i) = "KsepTotal_fr"
        ar3_power(i) = .ESP.Transform_eff_fr: ar4_names_power(i) = "Transform_eff_fr"
        i = i + 1
        ar1(i) = .ESP.KsepNat_fr:  ar2(i) = "KsepNat_fr"
        ar3_power(i) = .ESP.dPower_protector_W: ar4_names_power(i) = "dPower_protector_W"
        i = i + 1
        ar1(i) = .ESP.KSepGasSep_fr:   ar2(i) = "KSepGasSep_fr "
        ar3_power(i) = .ESP.dPower_GasSep_W: ar4_names_power(i) = "dPower_GasSep_W"
        ' параметры температуры
        i = i + 1
        ar1(i) = .Tsurf_C:  ar2(i) = "Tsurf_C"
        ar3_power(i) = .ESP.ESPpump.PowerFluid_Wt: ar4_names_power(i) = "PowerFluid_Wt"
        i = i + 1
        ar1(i) = .Twh_C:  ar2(i) = "Twh_C"
        ar3_power(i) = .ESP.ESPpump.PowerESP_Wt: ar4_names_power(i) = "PowerESP_Wt"
        i = i + 1
        ar1(i) = .Tdis_C:  ar2(i) = "Tdis_C"
        ar3_power(i) = .ESP.ESPpump.EffiencyESP_d: ar4_names_power(i) = "EffiencyESP_d"
        i = i + 1
        ar1(i) = .Tint_C:  ar2(i) = "Tint_C"
        ar3_power(i) = eff_ESP_total_d: ar4_names_power(i) = "eff_ESP_total_d"
        i = i + 1
        ar1(i) = .Tbh_C:  ar2(i) = "Tbh_C"
        
        
'        ar1 = Array(0, "", .Pline_atma, .Pbuf_atma, .Pdis_atma, .Pint_atma, .Pwf_atma, "Давления атм, линейное, буферное, на выкиде насоса, на приеме насоса, забойное")
'        ar2 = Array(0, "", .Tsurf_C, .Twh_C, .Tdis_C, .Tint_C, .Tbh_C, "Температура С, на поверхности, буферное, на выкиде насоса, на приеме насоса, забойное")
'        ar3 = Array(0, "", .ESP.Power_CS_calc_W, .ESP.I_A, .ESP.Load_fr, .ESP.KsepTotal_fr, .ESP.ESPpump.EffiencyESP_d, "Мощность, Ток, Загрузка, Сепарация ")
        
        Select Case FirsrCol
            Case 0
                ar1(0) = .Pline_atma: ar2(0) = "Plin"
            Case 1
                ar1(0) = .Pwf_atma:   ar2(0) = "Pwf"
            Case 2
                ar1(0) = .Qliq_scm3day:   ar2(0) = "Qliq"
        End Select
        Select Case type_out
            Case 0
                Well_out_arr = Array(ar1, ar2)
            Case 1
                Well_out_arr = Array(ar3_power, ar4_names_power)
            Case 2
                Well_out_arr = eff_ESP_total_d
            Case 3
                Well_out_arr = well.ESP.ESPpump.EffiencyESP_d
            Case 4
                Well_out_arr = well.ESP.ESPpump.PowerESP_Wt
            Case 5
                Well_out_arr = well.ESP.ESPpump.Head_m
            Case 6
                Well_out_arr = well.ESP.efficency_ESP_total_d
        End Select
        
    ' можно еще добавить сюда вывод кривых распределения давления и температуры по стволу и еще 4 параметров  (потом)
    End With
End Function
Public Function Well_Plin_Pwf_atma(Pwf_atma, Q_m3Day, fw_perc, _
                      Optional ByVal Pcas_atma As Double = 10, _
                      Optional ByVal wellStr As String = "Hperf_m:2000,00000;Hpump_m:1800,00000;Udl_m:0,00000;dCas_mm:150,00000;dTub_mm:72,00000;dchoke_mm:15,00000;roughness_m:0,00010;Tbh_C:85,00000;Twh_C:25,00000;", _
                      Optional ByVal PVTStr As String = "gamma_gas:0,600;gamma_oil:0,860;gamma_wat:1,000;rsb_m3m3:100,000;Rp_m3m3:-1,000;Pb_atma:-1,000;Tres_C:90,000;Bob_m3m3:-1,000;Muob_cP:-1,000;PVTcorr:0;Ksep_fr:0,000;PKsep_atma:-1,000;TKsep_C:-1,000;", _
                      Optional ByVal ESPStr As String = "ESP_ID:0,00000;HeadNom_m:2000,00000;ESPfreq_Hz:50,00000;ESP_U_V:1000,00000;MotorPowerNom_kW:30,00000;Tint_C:85,00000;Tdis_C:25,00000;", _
                      Optional ByVal HydrCorr As H_CORRELATION = 0, _
                      Optional ByVal Ksep_fr As Double = 0, _
                      Optional ByVal Kdegr_d As Double = 0, _
                      Optional ByVal param_num As Integer = 0, _
                      Optional ByVal type_out As Integer = 0, _
                      Optional ByVal existence_emulsion As Boolean = False)
' функция расчета линейного давления скважины по забойному
    Dim well As CWell
    Set well = well_InitData(Q_m3Day, fw_perc, Pcas_atma, _
                             wellStr, PVTStr, ESPStr, AnsariCor, Ksep_fr, Kdegr_d)
    Call well.Calc_Plin_Pwf_atma(SetPT(Pwf_atma, well.Tbh_C), with_Emulsion:=existence_emulsion)          ' проведем расчет
    ' в качестве результата выведем ряд расчитанных значений давления
    Well_Plin_Pwf_atma = Well_out_arr(well, 0, type_out)
End Function
Public Function Well_calc(Pwf_atma, Q_m3Day, fw_perc, _
                      Optional ByVal Pcas_atma As Double = 10, _
                      Optional wellStr As String = WELL_DEFAULT, _
                      Optional PVTStr As String = PVT_DEFAULT, _
                      Optional ESPStr As String = ESP_DEFAULT, _
                      Optional ByVal HydrCorr As H_CORRELATION = 0, _
                      Optional ByVal Ksep_fr As Double = 0, _
                      Optional ByVal Kdegr_d As Double = 0, _
                      Optional ByVal param_num As Integer = 0)
' функция расчета прозвольного параметра скважины по забойному давлению
    Dim well As CWell
    Dim ar
    Set well = well_InitData(Q_m3Day, fw_perc, Pcas_atma, _
                             wellStr, PVTStr, ESPStr, AnsariCor, Ksep_fr, Kdegr_d)
    Call well.Calc_Plin_Pwf_atma(SetPT(Pwf_atma, well.Tbh_C))         ' проведем расчет
    ' в качестве результата выведем ряд расчитанных значений давления
    ar = Well_out_arr(well, 0)
    Well_calc = Array(ar(0)(param_num), ar(1)(param_num))
    If param_num > 0 Then
        Well_calc = Array(ar(0)(param_num), ar(1)(param_num))
    Else
        Well_calc = ar
    End If
End Function
Public Function Well_Pwf_Plin_atma(Plin_atma, Q_m3Day, fw_perc, _
                      Optional ByVal Pcas_atma As Double = 10, _
                      Optional wellStr As String = WELL_DEFAULT, _
                      Optional PVTStr As String = PVT_DEFAULT, _
                      Optional ESPStr As String = ESP_DEFAULT, _
                      Optional ByVal HydrCorr As H_CORRELATION = 0, _
                      Optional ByVal Ksep_fr As Double = -1, _
                      Optional ByVal Psep_atma As Double = 40, _
                      Optional ByVal Tsep_C As Double = 40, _
                      Optional ByVal Kdegr_d As Double = 0, _
                      Optional ByVal param_num As Integer = 0)
' функция расчета забойного давления скважины по линейному
' на основе устьевых параметров работы скважины
    Dim well As CWell
    Dim ar
    Set well = well_InitData(Q_m3Day, fw_perc, Pcas_atma, _
                             wellStr, PVTStr, ESPStr, AnsariCor, Ksep_fr, Kdegr_d)
    ' при расчете сверху вниз задаем какие то значения сепарации наугад
    Call well.setKseptotal_d(fullyManual, Ksep_fr, 0, Psep_atma, Tsep_C)
    Call well.Calc_Pwf_Plin_atma(Plin_atma, well.Tbh_C, fast:=True, CalcCasing:=True)
    ' проведем расчет
    ' в качестве результата выведем ряд расчитанных значений давления
    ar = Well_out_arr(well, 1)
    If param_num > 0 Then
        Well_Pwf_Plin_atma = Array(ar(0)(param_num), ar(1)(param_num))
    Else
        Well_Pwf_Plin_atma = ar
    End If
End Function

Public Function Nodal_Qliq_scm3day(PI_m3dayatm, Plin_atma, fw_perc, _
                      Optional ByVal Pres_atma = 250, _
                      Optional ByVal Pcas_atma As Double = 10, _
                      Optional wellStr As String = WELL_DEFAULT, _
                      Optional PVTStr As String = PVT_DEFAULT, _
                      Optional ESPStr As String = ESP_DEFAULT, _
                      Optional ByVal HydrCorr As H_CORRELATION = 0, _
                      Optional ByVal Ksep_fr As Double = 0, _
                      Optional ByVal Kdegr_d As Double = 0, _
                      Optional ByVal param_num As Integer = 0)
' расчет узлового анализа по скважине
    Dim PS As New CProdSystem
    Dim well As CWell
    Set well = well_InitData(0, fw_perc, Pcas_atma, _
                             wellStr, PVTStr, ESPStr, AnsariCor, Ksep_fr, Kdegr_d)
    PS.Reservoir.InitProp Pres_atma, well.Fluid.Pb_atma, fw_perc
    PS.Reservoir.PI_m3dayatm = PI_m3dayatm
    Set PS.well = well
    Set PS.Fluid = well.Fluid
    
    PS.Calc_Nodal_PI Plin_atma
        
    Call well.Calc_Plin_Pwf_atma(SetPT(PS.Pwf_atma, well.Tbh_C))         ' проведем расчет
    ' в качестве результата выведем ряд расчитанных значений давления
    Dim ar
    ar = Well_out_arr(well, 0)
    If param_num > 0 Then
        Nodal_Qliq_scm3day = Array(ar(0)(param_num), ar(1)(param_num))
    Else
        Nodal_Qliq_scm3day = Well_out_arr(well, 2)
    End If
End Function
