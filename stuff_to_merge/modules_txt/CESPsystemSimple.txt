Public Function calc_separation(P_atma As Double, t_c As Double)
' расчет сепарации и соответствующих параметров флюидов
' должен вызываться незавимо от расчета УЭЦН, так как при расчете сверху вниз
'   необходимо заранее рассчитать параметры флюида выше насоса, чтобы правильно посчитать НКТ
' P_atma  - давление при которой идет сепарация
' T_C - температура при которой идет сепарация
    Select Case isManualSeparationCalc
        Case byCorrealation
            ' все по сепарации по корреляциям
            Psep_atma = P_atma
            Tsep_C = t_c
            Call p_Fluid.Calc_PVT(Psep_atma, Tsep_C)
            With p_Fluid
                KsepNat_fr = unf_calc_natural_separation(dIntake_mm / 1000, dCas_mm / 1000, _
                                            .Qliq_scm3day, .Qgas_scm3day, .Bo_m3m3, .Bg_m3m3, _
                                            .sigma_o_Nm, .RhoOil_sckgm3, .rhoGas_sckgm3, .wc_perc)
                                    
            End With
        Case pressureManual
            ' берем значение ручное, давления как есть
             Call p_Fluid.Calc_PVT(Psep_atma, Tsep_C)
             With p_Fluid
                KsepNat_fr = unf_calc_natural_separation(dIntake_mm / 1000, dCas_mm / 1000, _
                                            .Qliq_scm3day, .Qgas_scm3day, .Bo_m3m3, .Bg_m3m3, _
                                            .sigma_o_Nm, .RhoOil_sckgm3, .rhoGas_sckgm3, .wc_perc)
                                    
             End With
        Case valueManual
            ' берем и коэффициенты и давления как было задано при установке значений
            Psep_atma = P_atma
            Tsep_C = t_c
            Call p_Fluid.Calc_PVT(Psep_atma, Tsep_C)
        Case fullyManual
            ' берем и коэффициенты и давления как было задано при установке значений
            Call p_Fluid.Calc_PVT(Psep_atma, Tsep_C)
        
    End Select
    
    ' потом, когда будет расчет газосепаратора - то надо будет разделить алгоритмы
    ' ручного задания и задания с использованием газосепаратора
    If isGasSep Then
        KSepGasSep_fr = KSepGasSep_fr
    Else
        KSepGasSep_fr = 0
    End If
    
    Set FluidTub = p_Fluid.Clone()
    Call FluidTub.ModAfterSeparation(Psep_atma, Tsep_C, KsepTotal_fr, GasGoesIntoSolution)
    Set FluidCas = p_Fluid.Clone() ' тут надо будет доделать чтобы считать затруб с газом
    FluidCas.Qgfree_scm3day = Fluid.QgasInSitu_scm3day * KsepTotal_fr
    Set FluidGasSep = p_Fluid.Clone()
    Call FluidGasSep.ModAfterSeparation(Psep_atma, Tsep_C, KSepGasSep_fr, GasGoesIntoSolution)
    
    Set ESPpump.Fluid = FluidTub
    
End Function
Public Function calc_ESPsys(P_atma As Double, Tin_C As Double, _
                            Optional ByVal Tdis_C As Double = -1, _
                            Optional ByVal CalcFromIntake As Boolean = True, _
                            Optional ByVal with_Emulsion As Boolean = False, _
                             Optional ByVal cor_Emulsion As Double = 0) As PTtype
' функция расчета параметров работы УЭЦН
'
' P_atma - давление начала расчета (давление на приеме если CalcFromIntake = True)
' Tin_C - температура на приеме насоса
' Tdis_C - температура на выкиде насоса. если =-1 то рассчитывается при CalcFromIntake = True
' CalcFromIntake - флаг показывает ведется ли расчет от давления на приеме или от выкида
'                  при расчете сверху вниз температура должна быть задана с двух сторон
'                  и сепарация будет учитываться в соответствии с параметрами заданными вручную
    
    '1. расчет распределения давления в ЭЦН
    Call ESPpump.Calc_ESP(P_atma, Tin_C, Tdis_C, CalcFromIntake, , with_Emulsion, cor_Emulsion)
    '2. Оценим мощность на валу
    Power_shaft_W = ESPpump.PowerESP_Wt + dPower_protector_W + dPower_GasSep_W
    '3. Оценим электрическую мощность потребляемую ПЭД
    ' Power_motor_W = Power_shaft_W / motor_eff_fr
    Power_motor_W = Power_shaft_W / motor_eff_fr ' активная потребляемая мощность двигателем
    '4. оценим потери в кабеле - для этого найдем примерно ток двигателя
    I_A = Power_motor_W / 1.732050808 / U_V / cos_phi
    'I_A = Power_motor_W / 1.732050808 / 400 / 70
    '5 по данным тока и сопротивления кабеля найдем падение напряжения в кабеле
    cable_dU_V = I_A * cable_R_Omkm * HmesTop_m / 1000
    '6 по падению напряжения оценим мощность рассеиваемую кабелем в тепло. учитываем 3 жилы. Тут учитывается полный ток и активный и реактивный
    'cable_dPower_W = 3 * cable_dU_V * I_A
    
    cable_dPower_W = 3 * cable_dU_V * I_A * cos_phi ' только активные потери
    'cable_dPower_W = 3 * cable_R_Omkm * HmesTop_m / 1000 * I_A ^ 2 * cos_phi
    '7 найдет мощность электрическую на входе в кабель
    cable_Power_W = Power_motor_W + cable_dPower_W
    'cable_Power_W = Power_motor_W / cos_phi + cable_dPower_W ' с учетом полной мощности двигателя
    '8. оценим потери мощность в трансформаторе и станции управления
    'dPower_transform_W = (1 - CS_eff_fr) * cable_Power_W
    dPower_transform_W = cable_Power_W / Transform_eff_fr * (1 - Transform_eff_fr)
    
    'dPower_CS_W = (1 - Transform_eff_fr) * (cable_Power_W + dPower_transform_W)
    dPower_CS_W = (cable_Power_W + dPower_transform_W) / CS_eff_fr * (1 - CS_eff_fr)
    '9. оценим общую потребляемую мощность УЭЦН
    Power_CS_teor_calc_W = cable_Power_W + dPower_transform_W + dPower_CS_W
    '10. внесем в мощность поправку на деградацию по мощности
    Power_CS_calc_W = Power_CS_teor_calc_W * (1 - power_degr_fr)
    ' 11 рассчитаем загрузку двигателя, для этого получим номинальную мощность на данной частоте
    
    PowNom_corr_W = Power_nom_W * ESPpump.freq_Hz / Freq_nom_Hz
    Load_fr = Power_shaft_W / PowNom_corr_W
    '  мощность при данной частоте, т.е. PowNom_corr_W
    ' Power_nom_W - номинальная мощность ПЭД, т.е. полезная механическая мощность на валу по паспорту
    calc_ESPsys = SetPT(ESPpump.Pdis_atma, ESPpump.Tdis_C)
    efficency_ESP_total_d = ESPpump.PowerFluid_Wt / Power_CS_teor_calc_W
    
'    degr = 1 - power_well_w /
    
'   Well_Power_w = Array(power_well_w, degr)
'
End Function

 Public Sub SetKseptotal(calc_method As SEPAR_CALC_METHOD, KsepNat_fr As Double, _
                         Optional KsepGS_fr As Double = 0, _
                         Optional Psep_atma As Double = -1, _
                         Optional Tsep_C As Double = -1)
 ' устанавливает значение сепарации для уэцн вручную
 ' сначала установим значение газосепаратора - оно всегда одинаково для всех вариантов
    
    If KsepGS_fr > 0 And KsepGS_fr <= 1 Then
        isGasSep = True
    Else
       isGasSep = False
    End If
    KSepGasSep_fr = KsepGS_fr
    
    Me.Psep_atma = Psep_atma
    Me.Tsep_C = Tsep_C
    Me.KsepNat_fr = KsepNat_fr
    
    ' установим метод расчета
    ' но так чтобы он не противоречил введенным данным
     If KsepNat_fr >= 0 And KsepNat_fr <= 1 And PTsep.P_atma > 0 And PTsep.t_c > 0 Then
        ' если ввели хорошие параметры то можно применить любой метод расчета
         isManualSeparationCalc = calc_method
     ElseIf KsepNat_fr >= 0 And KsepNat_fr <= 1 And PTsep.P_atma <= 0 Then
        ' плохое значение давления не позволяет полностью ручной метод
        If calc_method <> fullyManual Then
            isManualSeparationCalc = calc_method
        Else
            isManualSeparationCalc = valueManual
        End If
     ElseIf (KsepNat_fr < 0 Or KsepNat_fr > 1) And (PTsep.P_atma > 0 And PTsep.t_c > 0) Then
        ' плохая сепарация, но хорошие давления
         If calc_method = byCorrealation Or calc_method = pressureManual Then
            isManualSeparationCalc = calc_method
         Else
            isManualSeparationCalc = pressureManual
        End If
     Else
         ' плохая сепарация не позволяет ручной расчет
         isManualSeparationCalc = byCorrealation
     End If
     
 End Sub
